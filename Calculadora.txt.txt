import java.util.Scanner;

public class Calculadora {
    public static void main(String[] args) {
        Scanner scanner = new Scanner (System.in);
       
        System.out.println("Digite o primeiro numero: ");
        double a = scanner.nextDouble();
        scanner.nextLine();
       
        System.out.println("Digite o segundo numero: ");
        double b = scanner.nextDouble();
        scanner.nextLine();
       
        System.out.println("Digite a operacao:");
        System.out.println("1 para adicao");
        System.out.println("2 para subtracao");
        System.out.println("3 para multiplicacao");
        System.out.println("4 para divisao");
        int opc = scanner.nextInt();
        scanner.nextLine();
       
        if (opc == 1){
            System.out.println("O resultado e: " + (a + b));
        }
        if (opc == 2){
            System.out.println("O resultado e: " + (a - b));
        }
        if (opc == 3){
            System.out.println("O resultado e: " + (a * b));
        }
        if (opc == 4){
            System.out.println("O resultado e: " + (a / b));
        }
        if (opc < 0 || opc > 4){
            System.out.println("Numero errado, tente novamente.");
        }
        scanner.close();
       
    }
}
