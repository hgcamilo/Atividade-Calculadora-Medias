package fag;

import java.util.Scanner;

public class Principal {

	public static void main(String[] args) {
		
		Scanner scanner = new Scanner(System.in);
		
		String nome;
		double numero1 = 0;
		double numero2 = 0;
		double numero3 = 0;
		
		for(int i=1; i <=3; i++){
			System.out.println("Digite o nome do aluno");
			nome = scanner.nextLine();
			
			System.out.println("Digite a primeira nota");
			numero1 = scanner.nextDouble();
			scanner.nextLine();
			
			System.out.println("Digite a segunda nota");
			numero2 = scanner.nextDouble();
			scanner.nextLine();
			
			System.out.println("Digite a terceira nota");
			numero3 = scanner.nextDouble();
			scanner.nextLine();
			
			double media = calcularmedia(numero1, numero2, numero3);
			exibirResultado(nome, media, calcularaprovacao(media));
			
		}
		
		
		scanner.close();
	}
	
	public static double calcularmedia(double numero1, double numero2, double numero3) {
		return (numero1 + numero2 + numero3) / 3;
		}
	
	public static String calcularaprovacao(double calcularmedia) {
		 if(calcularmedia >= 7) {
			return "Aprovado";
		} else {
			return "Reprovado";
		}
	}
	
	public static void exibirResultado(String nome, double calcularmedia, String calcularaprovacao) {
		System.out.println("nome do aluno: " + nome);
		System.out.println("media do aluno: " + calcularmedia);
		System.out.println("status: " + calcularaprovacao);
		
	}
	
}
